from tkinter import *

# Ініціалізація головного вікна
root = Tk()
root.title("Калькулятор")
root.geometry("400x500")
root.configure(bg="#383838")
root.resizable(False, False)
                    
# Поле для введення виразу
example = Entry(root, font='Arial 20', justify='right')
example.grid(row=0, column=0, columnspan=4, pady=10, padx=10)
                    
# Функція для обробки натискання кнопок
def button_click(value):
    current = example.get()
    example.delete(0, END)
    example.insert(0, current + value)
                    
# Функція для обчислення результату
def calc():
    try:
        ex = example.get()
        result = eval(ex)
        example.delete(0, END)
        example.insert(0, str(result))
    except Exception:
        example.delete(0, END)
        example.insert(0, "Error")
                    
# Функція для очищення поля
def clear():
    example.delete(0, END)
                    
# Створення кнопок
buttons = [
    ('7', 1, 0), ('8', 1, 1), ('9', 1, 2), ('/', 1, 3),
    ('4', 2, 0), ('5', 2, 1), ('6', 2, 2), ('*', 2, 3),
    ('1', 3, 0), ('2', 3, 1), ('3', 3, 2), ('-', 3, 3),
    ('C', 4, 0), ('0', 4, 1), ('.', 4, 2), ('+', 4, 3),
    ('=', 5, 0, 4)  # Кнопка "=" займає весь ряд
]
                    
for (text, row, col, *span) in buttons:
    if text == '=':
        Button(root, text=text, fg='white', activebackground='#247de3', bg='#383838', font='Arial 20', command=calc).grid(row=row, column=col, columnspan=span[0], sticky="nsew")
    elif text == 'C':
        Button(root, text=text, fg='white', activebackground='#247de3', bg='#383838', font='Arial 20', command=clear).grid(row=row, column=col, sticky="nsew")
    else:
        Button(root, text=text, fg='white', activebackground='#247de3', bg='#383838', font='Arial 20', command=lambda t=text: button_click(t)).grid(row=row, column=col, sticky="nsew")
                    
# Налаштування розмірів колонок і рядків
for i in range(4):
    root.columnconfigure(i, weight=1)
for i in range(6):
    root.rowconfigure(i, weight=1)
                    
root.mainloop()
